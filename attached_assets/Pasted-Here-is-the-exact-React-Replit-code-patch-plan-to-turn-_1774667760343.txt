4. Add reusable creator avatar component
client/src/components/concierge/CreatorAvatar.tsx
import creatorAvatar from "@/assets/creator/creator-avatar.webp";
import clsx from "clsx";

type CreatorAvatarProps = {
  size?: "sm" | "md" | "lg";
  live?: boolean;
  speaking?: boolean;
  className?: string;
};

export function CreatorAvatar({
  size = "md",
  live = false,
  speaking = false,
  className,
}: CreatorAvatarProps) {
  const sizeClass =
    size === "sm"
      ? "h-10 w-10"
      : size === "lg"
      ? "h-20 w-20"
      : "h-14 w-14";

  return (
    <div
      className={clsx(
        "relative rounded-full p-[2px] bg-gradient-to-br from-[#F4A62A] via-[#D8A12F] to-[#2A2E4A]",
        live && "e360-avatar-live",
        speaking && "e360-avatar-speaking",
        className
      )}
    >
      <div className="relative overflow-hidden rounded-full bg-[#0B1020]">
        <img
          src={creatorAvatar}
          alt="Elevate360 founder concierge"
          className={clsx(sizeClass, "rounded-full object-cover")}
        />
      </div>

      <span className="absolute bottom-0 right-0 h-3.5 w-3.5 rounded-full border-2 border-[#0B1020] bg-[#F4A62A]" />
    </div>
  );
}

10. Patch the consultation card CTA behavior

When a user clicks a session, also sync concierge mode:

const handleSessionSelect = (mode: ConciergeModeKey, bookingUrl: string) => {
  setActiveSessionMode(mode);
  window.dispatchEvent(
    new CustomEvent("e360:set-concierge-mode", { detail: { mode } })
  );
  window.open(bookingUrl, "_blank");
};

In AIConcierge.tsx, listen for it:

useEffect(() => {
  const handler = (event: Event) => {
    const custom = event as CustomEvent<{ mode: ConciergeModeKey }>;
    if (custom.detail?.mode) setMode(custom.detail.mode);
  };

  window.addEventListener("e360:set-concierge-mode", handler as EventListener);
  return () => window.removeEventListener("e360:set-concierge-mode", handler as EventListener);
}, []);

This keeps the booking cards and concierge panel synchronized.
