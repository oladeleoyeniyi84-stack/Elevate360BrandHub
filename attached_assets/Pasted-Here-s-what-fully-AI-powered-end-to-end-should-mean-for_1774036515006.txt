9) OpenAI client wrapper
server/ai/client.ts
export async function callOpenAI(messages: Array<{ role: string; content: string }>) {
  const apiKey = process.env.OPENAI_API_KEY;
  if (!apiKey) throw new Error("Missing OPENAI_API_KEY");

  const response = await fetch("https://api.openai.com/v1/chat/completions", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${apiKey}`,
    },
    body: JSON.stringify({
      model: "gpt-4.1-mini",
      temperature: 0.7,
      messages,
    }),
  });

  if (!response.ok) {
    const text = await response.text();
    throw new Error(`OpenAI error: ${text}`);
  }

  const data = await response.json();
  return data.choices?.[0]?.message?.content ?? "";
}

11) Knowledge retrieval logic

Start simple: use DB text search and pass best documents into the prompt.

server/ai/retrieval.ts
import { db } from "../db";
import { knowledgeDocuments } from "../db/schema";
import { ilike, and, eq } from "drizzle-orm";

export async function searchKnowledge(query: string) {
  const q = `%${query}%`;

  const docs = await db
    .select()
    .from(knowledgeDocuments)
    .where(and(
      eq(knowledgeDocuments.published, true),
      ilike(knowledgeDocuments.textContent, q)
    ))
    .limit(5);

  return docs;
}

export function formatKnowledgeContext(docs: Array<{ title: string; textContent: string | null }>) {
  return docs.map((doc, index) => {
    return `Document ${index + 1}: ${doc.title}\n${doc.textContent ?? ""}`;
  }).join("\n\n");
}

Later, replace this with embeddings/vector search.

12) Concierge orchestration
server/ai/concierge.ts
import { callOpenAI } from "./client";
import { conciergeSystemPrompt } from "./prompts";
import { searchKnowledge, formatKnowledgeContext } from "./retrieval";

export async function runConcierge(userMessage: string) {
  const docs = await searchKnowledge(userMessage);
  const context = formatKnowledgeContext(docs);

  const reply = await callOpenAI([
    { role: "system", content: conciergeSystemPrompt },
    {
      role: "system",
      content: `Relevant brand knowledge:\n${context || "No direct knowledge found."}`,
    },
    { role: "user", content: userMessage },
  ]);

  return {
    reply,
    docsUsed: docs.map((d) => d.title),
  };
}
13) Lead scoring logic
server/ai/leadScoring.ts
type LeadInput = {
  inquiryType?: string;
  budgetRange?: string;
  timeline?: string;
  urgency?: string;
  email?: string;
};

export function scoreLead(input: LeadInput): number {
  let score = 0;

  if (input.email) score += 10;
  if (input.inquiryType && ["service", "booking", "partnership"].includes(input.inquiryType)) score += 20;
  if (input.budgetRange && !/low|unknown/i.test(input.budgetRange)) score += 20;
  if (input.timeline && /immediate|this week|this month/i.test(input.timeline)) score += 20;
  if (input.urgency && /high|urgent/i.test(input.urgency)) score += 15;

  return Math.min(score, 100);
}

export async function apiGet<T>(url: string): Promise<T> {
  const res = await fetch(url, {
    credentials: "include",
    headers: { "Accept": "application/json" },
  });

  if (!res.ok) throw new Error(`GET ${url} failed`);
  return res.json();
}

export async function apiPost<T>(url: string, body: unknown): Promise<T> {
  const res = await fetch(url, {
    method: "POST",
    credentials: "include",
    headers: {
      "Content-Type": "application/json",
      "Accept": "application/json",
    },
    body: JSON.stringify(body),
  });

  if (!res.ok) throw new Error(`POST ${url} failed`);
  return res.json();
}

import { useState } from "react";
import { apiPost } from "../../lib/api";

type ChatItem = {
  role: "user" | "assistant";
  content: string;
};

export default function ConciergeWidget() {
  const [message, setMessage] = useState("");
  const [loading, setLoading] = useState(false);
  const [chat, setChat] = useState<ChatItem[]>([
    {
      role: "assistant",
      content: "Welcome to Elevate360Official. Ask about services, apps, media, support, or booking.",
    },
  ]);

  async function sendMessage() {
    if (!message.trim()) return;
    const userText = message.trim();

    setChat((prev) => [...prev, { role: "user", content: userText }]);
    setMessage("");
    setLoading(true);

    try {
      const data = await apiPost<{ ok: boolean; reply: string }>("/api/ai/chat", {
        message: userText,
      });

      setChat((prev) => [...prev, { role: "assistant", content: data.reply }]);
    } catch {
      setChat((prev) => [
        ...prev,
        {
          role: "assistant",
          content: "I’m sorry — the AI concierge is temporarily unavailable. Please try again shortly.",
        },
      ]);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="rounded-2xl border border-white/10 bg-white/5 p-4 backdrop-blur">
      <h3 className="mb-3 text-lg font-semibold text-white">Elevate360 AI Concierge</h3>

      <div className="mb-3 max-h-96 space-y-3 overflow-y-auto rounded-xl bg-black/20 p-3">
        {chat.map((item, idx) => (
          <div
            key={idx}
            className={item.role === "user" ? "text-right" : "text-left"}
          >
            <div className="inline-block max-w-[85%] rounded-xl px-3 py-2 text-sm text-white bg-white/10">
              {item.content}
            </div>
          </div>
        ))}
      </div>

      <div className="flex gap-2">
        <input
          value={message}
          onChange={(e) => setMessage(e.target.value)}
          placeholder="Ask Elevate360 AI..."
          className="flex-1 rounded-xl border border-white/10 bg-black/20 px-3 py-2 text-white outline-none"
          onKeyDown={(e) => e.key === "Enter" && sendMessage()}
        />
        <button
          onClick={sendMessage}
          disabled={loading}
          className="rounded-xl bg-yellow-500 px-4 py-2 font-medium text-black disabled:opacity-60"
        >
          {loading ? "..." : "Send"}
        </button>
      </div>
    </div>
  );
}

  } catch (error) {

import { Request, Response, NextFunction } from "express";

export function errorHandler(err: unknown, _req: Request, res: Response, _next: NextFunction) {
  console.error(err);
  const message = err instanceof Error ? err.message : "Unknown server error";
  res.status(500).json({ ok: false, error: message });
}
