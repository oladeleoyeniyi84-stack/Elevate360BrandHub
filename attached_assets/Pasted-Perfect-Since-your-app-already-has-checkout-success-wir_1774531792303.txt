  return (
    <Switch>
      <Route path="/" component={Home} />
      <Route path="/links" component={Links} />
      <Route path="/dashboard" component={Dashboard} />
      <Route path="/blog" component={Blog} />

    </Switch>

          <AnnouncementBanner />
          <WhatsAppButton />

          <CookieBanner />

  );

function formatAmount(amount?: string | null) {
  if (!amount) return null;

  const numeric = Number(amount);
  if (Number.isNaN(numeric)) return amount;

  return new Intl.NumberFormat("en-US", {
    style: "currency",
    currency: "USD",
  }).format(numeric / 100);
}

function titleCase(value: string) {
  return value
    .replace(/[-_]/g, " ")
    .replace(/\b\w/g, (char) => char.toUpperCase());
}

Your existing Stripe flow already works with this

Right now your server redirects users to:

/checkout/success?session_id=...

That is fully compatible with the new page above.

So even if you change nothing else in the payment route, users will now see a polished brand-aligned thank-you page after every completed Stripe purchase.

Best upgrade for your checkout session route

In server/routes.ts, update the success URL inside POST /api/checkout/session from this style:

success_url: `${appUrl}/checkout/success?session_id={CHECKOUT_SESSION_ID}`,

to this stronger version:

success_url: `${appUrl}/checkout/success?source=purchase&session_id={CHECKOUT_SESSION_ID}`,

And if you already know the plan/product name and amount in that route, use:

success_url: `${appUrl}/checkout/success?source=purchase&plan=${encodeURIComponent(productName)}&amount=${unitAmount}&session_id={CHECKOUT_SESSION_ID}`,

Where:

productName is your offer/product label
unitAmount is Stripe cents, like 4900
Reuse the same page for all future transactions

You can now send users to either:

/checkout/success?source=purchase&session_id=...

or

/thank-you?source=booking&ref=BK-2049
/thank-you?source=message
/thank-you?source=order&ref=ORD-8841

Because both /checkout/success and /thank-you point to the same component.
